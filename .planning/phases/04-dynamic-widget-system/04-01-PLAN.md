    1. Class fields order: `private readonly vcr`, `private readonly destroyRef`, `private readonly injector` declared first (Vitest DI safety pattern from Phase 02 decisions — inject() fields before other fields).

    2. `readonly configs = input&lt;WidgetConfig&lt;object&gt;[]&gt;([])` — use input() signal, not @Input() decorator (SIG-01 project pattern).

    3. The config-change effect lives in constructor() — not ngOnInit or ngOnChanges:
    ```typescript
    constructor() {
      effect(() => {
        const cfgs = this.configs();
        this.vcr.clear();
        for (const cfg of cfgs) {
          this.renderOne(cfg);
        }
      });
      this.destroyRef.onDestroy(() => this.vcr.clear());
    }
    ```

    4. renderOne() private method handles the switch:
    ```typescript
    private renderOne(cfg: WidgetConfig&lt;object&gt;): void {
      const ref = this.vcr.createComponent(cfg.component);
      for (const [key, binding] of Object.entries(cfg.inputs ?? {})) {
        const b = binding as InputBinding&lt;unknown&gt;;
        switch (b.type) {
          case 'static':
            ref.setInput(key, b.value);
            break;
          case 'observable':
            (b.value as Observable&lt;unknown&gt;)
              .pipe(takeUntilDestroyed(this.destroyRef))
              .subscribe((v) => ref.setInput(key, v));
            break;
          case 'signal':
            effect(() => ref.setInput(key, (b.value as Signal&lt;unknown&gt;)()), {
              injector: this.injector,
            });
            break;
        }
      }
      for (const [key, handler] of Object.entries(cfg.outputs ?? {})) {
        const outputRef = (ref.instance as Record&lt;string, { subscribe: (h: (v: unknown) => void) => void }&gt;)[key];
        if (outputRef?.subscribe) {
          outputRef.subscribe(handler as (v: unknown) => void);
        }
      }
    }
    ```
