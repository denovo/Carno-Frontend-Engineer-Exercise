```yaml
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  lint-and-test:
    name: Lint and Unit Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "23"
          cache: "npm"

      - name: Install dependencies
        run: npm ci

      - name: Format check (OXfmt)
        run: npm run format:check

      - name: Unit tests (Vitest via ng test)
        # Covers TOOL-03 (selectors), TOOL-04 (reducer), TOOL-05 (effects)
        # Uses ng test --watch=false via @angular/build:unit-test — bare vitest lacks @app/ alias
        run: npm test

  build:
    name: Build (Angular + Storybook)
    needs: lint-and-test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "23"
          cache: "npm"

      - name: Install dependencies
        run: npm ci

      - name: Build Angular app
        run: npm run build

      - name: Build Storybook (fails CI if stories break)
        run: npx storybook@8 build

      - name: Upload dist artifact for E2E job
        uses: actions/upload-artifact@v4
        with:
          name: dist
          path: dist/
          retention-days: 1

  e2e:
    name: Playwright E2E Tests
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "23"
          cache: "npm"

      - name: Install dependencies
        run: npm ci

      - name: Install Playwright Chromium browser
        run: npx playwright install --with-deps chromium

      - name: Download dist artifact
        uses: actions/download-artifact@v4
        with:
          name: dist
          path: dist/

      - name: Run Playwright E2E tests
        # playwright.config.ts uses process.env.CI to switch webServer to:
        # "npx serve dist/petello/browser -l 4200 --no-clipboard"
        run: npx playwright test
        env:
          CI: "true"

      - name: Upload Playwright report (always — for debugging failures)
        uses: actions/upload-artifact@v4
        if: ${{ !cancelled() }}
        with:
          name: playwright-report
          path: playwright-report/
          retention-days: 30
```
